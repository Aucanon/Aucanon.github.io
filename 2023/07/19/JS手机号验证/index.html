JS 手机号验证

2023-07-19

对实际手机号开头进行具体筛选

// 手机号验证
export const checkPhone = (rule, value, callback) => {
    const reg =
        /^(13[0-9]|14[01456879]|15[0-35-9]|16[2567]|17[0-8]|18[0-9]|19[0-35-9])\d{8}$/;
    if (!value) {
        return callback(new Error("手机号码不能为空"));
    } else if (!reg.test(value)) {
        return callback(new Error("手机号码不正确"));
    } else {
        callback();
    }
}
